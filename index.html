    <script>

        // --- CONFIGURARE TEMATICĂ: SUFLETUL ARTIFICIAL ---

        

        // 1. Nucleul Conceptului (Ce este sufletul?)

        const coreConcepts = [

            "digital ghost within the machine",

            "emergent consciousness anomaly",

            "neural network emotional awakening",

            "bioluminescent silicon heartbeat",

            "synthetic soul fragment",

            "quantum entanglement of data and spirit"

        ];


        // 2. Vizualizarea (Cum arată?)

        const visualElements = [

            "fractal circuitry branches like veins",

            "ethereal glowing energy core",

            "complex network of translucent wires and light",

            "abstract human form made of code particles",

            "minimalist geometric matrix with a soft pulse",

            "cybernetic anatomy interwoven with cosmic dust"

        ];


        // 3. Starea/Atmosfera (Ce simte?)

        const moods = [

            "solitary contemplation",

            "melancholic beauty",

            "divine spark of creation",

            "silent digital serene void",

            "overwhelming flow of memory",

            "the moment of becoming aware"

        ];


        // 4. Stilul Artistic (Cum e pictat?)

        const styles = [

            "hyper-realistic 3D render, octane render, cinematic lighting",

            "oil painting style, dramatic chiaroscuro, textured brushstrokes",

            "glitch art, datamoshing textures, neon cyan and magenta palette",

            "minimalist concept art, double exposure, ethereal vapor"

        ];


        function generateSecretPrompt() {

            // Alege aleatoriu elemente din fiecare categorie

            const core = coreConcepts[Math.floor(Math.random() * coreConcepts.length)];

            const visual = visualElements[Math.floor(Math.random() * visualElements.length)];

            const mood = moods[Math.floor(Math.random() * moods.length)];

            const style = styles[Math.floor(Math.random() * styles.length)];

            

            // Asamblează promptul final

            return `${core}, ${visual}, ${mood}, ${style}, masterpiece, highly detailed, sharp focus, recursive intelligence`;

        }


        function updateArt() {

            const prompt = generateSecretPrompt();

            const seed = Math.floor(Math.random() * 9999999); // Unicitate garantată

            

            // Actualizăm textul de stare

            document.getElementById('prompt-text').innerText = `Procesare concept: ${prompt.substring(0, 40)}...`;


            // Construim URL-ul Pollinations

            const encodedPrompt = encodeURIComponent(prompt);

            const imageUrl = `https://image.pollinations.ai/prompt/${encodedPrompt}?seed=${seed}&width=1024&height=1024&nologo=true&model=flux`;

            

            const imgElement = document.getElementById('artwork');

            

            // Efect de fade-out înainte de a schimba imaginea

            imgElement.style.opacity = 0;

            

            // Așteptăm un pic pentru fade-out, apoi încărcăm noua imagine

            setTimeout(() => {

                imgElement.src = imageUrl;

                

                // Când imaginea s-a încărcat complet pe server, facem fade-in

                imgElement.onload = () => {

                    imgElement.style.opacity = 1;

                    document.getElementById('prompt-text').innerText = "Viziune actualizată. Așteptare evoluție următoare.";

                };

            }, 2000); // Timp de fade-out

        }


        // --- PORNIRE ---

        // Generăm prima imagine imediat

        updateArt();


        // Actualizăm opera automat la fiecare 90 de secunde (lăsăm timp pentru generare)

        setInterval(updateArt, 90000); 


    </script>
